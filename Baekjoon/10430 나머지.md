import Foundation

func operate() {
    guard let userInput = readLine()?.components(separatedBy: " ").compactMap({ Int($0) }).filter({$0 >= 2 && $0 <= 10000}) else {
        return operate()
    }
    print((userInput[0] + userInput[1]) % userInput[2])
    print(((userInput[0] % userInput[2]) + (userInput[1] + userInput[2])) % userInput[2])
    print(((userInput[0] * userInput[1])) % userInput[2])
    print(((userInput[0] % userInput[2]) * (userInput[1] % userInput[2]) % userInput[2]))
}

operate()
